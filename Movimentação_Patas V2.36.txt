void setup() {

  for(int i = 22; i<34; i++) pinMode(i , OUTPUT);//Portas para controlar Ponte H
  for(int i = 0; i<7; i++) pinMode(motores[i] , INPUT_PULLUP); //Portas 2 - 8 como entradas com resistor Pullup
  pinMode(8, OUTPUT);
  pinMode(9, OUTPUT);

  vw_set_rx_pin(pinRF);
  vw_setup(2000);   
  vw_rx_start();
  
}

void loop() {
  controle();
}

void controle(){
  if (vw_have_message()){
   vw_get_message(buf, &buflen);   
   memcpy(&pacote,&buf,buflen);
    switch (pacote.valor1) {
          case 'W':
            front();
          break;

          case 'S':
            front_auto();
          break;
          
          case 'A':
            left();
          break;
  
          case 'D':
            right();
          break;
  
          case 'C':
            parado(0);
          break;
          
          case'G':
            LED = !LED;
            digitalWrite(8, LED);
          break;

          case'B':
            CAMERA = !CAMERA;
            digitalWrite(9, CAMERA);
          break; 
    }
  }
}
